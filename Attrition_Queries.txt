1. Workforce Structural Ordering & Visibility
Sql
Copy code
-- 1.1 Order employees by Age (Ascending)
SELECT EmployeeID, Age, Department, YearsAtCompany
FROM Employee
ORDER BY Age ASC;

-- 1.2 Order employees by Department (Ascending)
SELECT EmployeeID, Department, Age, YearsAtCompany
FROM Employee
ORDER BY Department ASC;

-- 1.3 Order employees by YearsAtCompany (Ascending)
SELECT EmployeeID, YearsAtCompany, Department, Age
FROM Employee
ORDER BY YearsAtCompany ASC;

-- 1.4 Order employees by Age (Descending)
SELECT EmployeeID, Age, Department, YearsAtCompany
FROM Employee
ORDER BY Age DESC;

-- 1.5 Order employees by Department (Descending)
SELECT EmployeeID, Department, Age, YearsAtCompany
FROM Employee
ORDER BY Department DESC;

-- 1.6 Order employees by YearsAtCompany (Descending)
SELECT EmployeeID, YearsAtCompany, Department, Age
FROM Employee
ORDER BY YearsAtCompany DESC;
2. Workforce Experience & Cumulative Metrics
Sql
Copy code
-- 2.1 Total Years at Company (All Employees)
SELECT SUM(YearsAtCompany) AS Total_YearsAtCompany
FROM Employee;

-- 2.2 Total Companies Worked by all Employees
SELECT SUM(NumCompaniesWorked) AS Total_CompaniesWorked
FROM Employee;

-- 2.3 Total Age of Employees
SELECT SUM(Age) AS Total_Age
FROM Employee;

-- 2.4 Total Attrition (Yes)
SELECT COUNT(*) AS Total_Attrition
FROM Employee
WHERE Attrition = 'Yes';

-- 2.5 Total Years at Company (Medical Education Field)
SELECT SUM(YearsAtCompany) AS Total_Years_Medical
FROM Employee
WHERE EducationField = 'Medical';
3. Workforce Averages & Demographics
Sql
Copy code
-- 3.1 Average Age (All Employees)
SELECT AVG(Age) AS Avg_Age
FROM Employee;

-- 3.2 Average Years at Company (All Employees)
SELECT AVG(YearsAtCompany) AS Avg_Tenure
FROM Employee;

-- 3.3 Average Companies Worked (All Employees)
SELECT AVG(NumCompaniesWorked) AS Avg_CompaniesWorked
FROM Employee;

-- 3.4 Average Age by Department (Research & Development)
SELECT AVG(Age) AS Avg_Age_RnD
FROM Employee
WHERE Department = 'Research & Development';

-- 3.5 Average Years at Company (Male Employees)
SELECT AVG(YearsAtCompany) AS Avg_Tenure_Male
FROM Employee
WHERE Gender = 'Male';

-- 3.6 Average Age by Education Field (Life Sciences)
SELECT AVG(Age) AS Avg_Age_LifeSciences
FROM Employee
WHERE EducationField = 'Life Sciences';
4. Workforce Size & Distribution Analysis
Sql
Copy code
-- 4.1 Total Number of Employees
SELECT COUNT(*) AS Total_Employees
FROM Employee;

-- 4.2 Attrition Count (Yes/No)
SELECT Attrition, COUNT(*) AS Employee_Count
FROM Employee
GROUP BY Attrition;

-- 4.3 Employees per Department
SELECT Department, COUNT(*) AS Employee_Count
FROM Employee
GROUP BY Department
ORDER BY Employee_Count DESC;

-- 4.4 Employees from Marketing Education Field
SELECT COUNT(*) AS Marketing_Employees
FROM Employee
WHERE EducationField = 'Marketing';

-- 4.5 Married Employees
SELECT COUNT(*) AS Married_Employees
FROM Employee
WHERE MaritalStatus = 'Married';
